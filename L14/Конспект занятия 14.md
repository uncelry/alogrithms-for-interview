class Solution:
    def isValid(self, s: str) -> bool:
        
        # Dict for open-close brackets
        d = {
            '(' : ')',
            '{': '}',
            '[': ']'
        }
        
        # Stack of close brackets
        stack = []
        
        # Iterate through each element
        for c in s:
            # If this is open bracket, add it's close bracket to stack
            if c in d:
                stack.append(d[c])
            # Elif stack is empty and close bracket is current char, return False
            elif len(stack) == 0:
                return False
            # Elif last el in stack is current close bracket, pop it
            elif stack[-1] == c:
                stack.pop()
            # Else, if last bracket is not current close bracket, return False
            else:
                return False
            
        # If stack is not empty, return False
        return len(stack) == 0

```

    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:

        nge = [0] * n  # —Ä–µ–∑—É–ª—å—Ç–∞—Ç, –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é –≤—Å–µ 0

class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        stack = []
        
        for digit in num:
            while stack and k > 0 and stack[-1] > digit:
                stack.pop()
                k -= 1
            stack.append(digit)
        
        # If k > 0, remove remaining k digits from the end of the stack
        stack = stack[:-k] if k > 0 else stack
        
        # Remove leading zeros
        result = ''.join(stack).lstrip('0')
        
        # Handle edge case where result might be empty
        return result if result else '0'                

```

def is_correct(s):
  spisok = []
  m = {')':'(', '}':'{', ']':'['}
  
  
  for el in s:
    if el in '({[':
      spisok.append(el)
    elif not spisok:
      return False
    elif spisok[-1] == m[el]:
      spisok.pop()
    else:
      return False
      
  return len(spisok) == 0
