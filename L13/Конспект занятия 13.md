    total = sum(nums)
    
    # –ï—Å–ª–∏ —Å—É–º–º–∞ –Ω–µ—á—ë—Ç–Ω–∞—è ‚Äî —Ä–∞–∑–±–∏—Ç—å –Ω–∞ –¥–≤–µ —Ä–∞–≤–Ω—ã–µ —á–∞—Å—Ç–∏ –Ω–µ–ª—å–∑—è
    if total % 2 != 0:

def can_partition(nums):
    total = sum(nums)
    
    # –ï—Å–ª–∏ —Å—É–º–º–∞ –Ω–µ—á—ë—Ç–Ω–∞—è ‚Äî —Ä–∞–∑–±–∏—Ç—å –Ω–∞ –¥–≤–µ —Ä–∞–≤–Ω—ã–µ —á–∞—Å—Ç–∏ –Ω–µ–ª—å–∑—è

def change(amount, coins):
    n = len(coins)
    
    # dp[i][j] ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ø–æ—Å–æ–±–æ–≤ –Ω–∞–±—Ä–∞—Ç—å —Å—É–º–º—É j,
    # –∏—Å–ø–æ–ª—å–∑—É—è –ø–µ—Ä–≤—ã–µ i —Ç–∏–ø–æ–≤ –º–æ–Ω–µ—Ç
    dp = [[0] * (amount + 1) for _ in range(n + 1)]
    
    dp[0][0] = 1  # –æ–¥–∏–Ω —Å–ø–æ—Å–æ–± –Ω–∞–±—Ä–∞—Ç—å 0 ‚Äî –Ω–∏—á–µ–≥–æ –Ω–µ –±—Ä–∞—Ç—å
    
    for i in range(1, n + 1):
        dp[i][0] = 1  # —Å—É–º–º—É 0 –º–æ–∂–Ω–æ –Ω–∞–±—Ä–∞—Ç—å –æ–¥–Ω–∏–º —Å–ø–æ—Å–æ–±–æ–º
        for j in range(1, amount + 1):
            dp[i][j] = dp[i - 1][j]
            if j >= coins[i - 1]:
                dp[i][j] += dp[i][j - coins[i - 1]]
    
    return dp[n][amount]
    

def change(amount, coins):
    # dp[i] ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ø–æ—Å–æ–±–æ–≤ –Ω–∞–±—Ä–∞—Ç—å —Å—É–º–º—É i
    dp = [0] * (amount + 1)
    dp[0] = 1  # –æ–¥–∏–Ω —Å–ø–æ—Å–æ–± –Ω–∞–±—Ä–∞—Ç—å 0 ‚Äî –Ω–∏—á–µ–≥–æ –Ω–µ –±—Ä–∞—Ç—å
    
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] += dp[i - coin]
    
    return dp[amount]

```

def coinChange(self, coins: 'List[int]', amount: 'int') -> 'int':
        dp = [0] + [float('inf') for i in range(amount)]
        for i in range(1, amount+1):
            for coin in coins:
                if i - coin >= 0:
                    dp[i] = min(dp[i], dp[i-coin] + 1)
        if dp[-1] == float('inf'):
            return -1
        return dp[-1]

```
