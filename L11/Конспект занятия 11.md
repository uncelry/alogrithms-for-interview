```
class Solution:
    def rob(self, nums: List[int]) -> int:
        
        # We can rob house before previous + current, 
        # or we can don't rob current house and leave sum with previous house robbed
        
        # If there's only 1 house, we don't need DP
        if len(nums) == 1:
            return nums[0]
        
        # Create DP arr
        dp = [0 for i in range(len(nums))]
        
        # Base cases
        dp[0] = nums[0]
        dp[1] = max(nums[0], nums[1])
        
        # Count DP
        for i in range(2, len(nums)):
            dp[i] = max(dp[i - 1], dp[i - 2] + nums[i])
        
        # Return last dp element
        return dp[-1]
        

class Solution:
    def rob(self, nums: List[int]) -> int:

        for i in range(len(nums)):
            prev_prev = nums[i-2] if i-2 >= 0 else 0
            prev = nums[i-1] if i-1 >= 0 else 0
            nums[i] = max(nums[i] + prev_prev, prev)
        
        return nums[-1]
```

```
class Solution:
    def maxSubArray(self, nums):
        dp = [*nums]
        for i in range(1, len(nums)):
            dp[i] = max(nums[i], nums[i] + dp[i-1])
        return max(dp)

class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        dp = [0 for i in range(len(nums))]
        
        # First element is max for first element subarray
        dp[0] = nums[0]
        
        # Keep track of max
        total_max = dp[0]
        
        # Count DP
        for i in range(1, len(nums)):
            dp[i] = nums[i] + (dp[i - 1] if dp[i - 1] > 0 else 0)
            
            # Update max
            total_max = max(total_max, dp[i])
        
        # Result is in total_max
        return total_max

```
