```
class NumArray:

    def __init__(self, nums: List[int]):
        self.preSum = nums
        for i in range(len(nums)-1):
            self.preSum[i+1] += self.preSum[i]

    def sumRange(self, left: int, right: int) -> int:
        if left == 0: return self.preSum[right]
        return self.preSum[right] - self.preSum[left-1]
```

```
from collections import defaultdict
class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:
        
        prefix_sum = 0
        sums_dict = defaultdict(int)
        sums_dict[0] = 1
        cnt = 0
        
        for num in nums:
            prefix_sum += num
            cnt += sums_dict[prefix_sum - k]
            sums_dict[prefix_sum] += 1
        
        return cnt

```

```
class Solution(object):
    def findMaxLength(self, nums):
        count = 0
        max_length=0
        table = {0: 0}
        for index, num in enumerate(nums, 1):
            if num == 0:
                count -= 1
            else:
                count += 1
            
            if count in table:
                max_length = max(max_length, index - table[count])
            else:
                table[count] = index
        
        return max_length

```

```
class Solution:
    def trap(self, height: List[int]) -> int:
        
        # Build left-to-right maximums
        max_ltr = [0] * len(height)
        curr_max = 0
        for i, h in enumerate(height):
            curr_max = max(curr_max, h)
            max_ltr[i] = curr_max
        
        # Build right-to-left maximums
        max_rtl = [0] * len(height)
        curr_max = 0
        for i in range(len(height)-1,-1,-1):
            curr_max = max(curr_max, height[i])
            max_rtl[i] = curr_max
        
        # Sum min of maximums - current height
        res = 0
        for i in range(len(height)):
            res += min(max_ltr[i], max_rtl[i]) - height[i]
        
        # Return result sum
        return res

```

```
    def findBuildings(self, heights: List[int]) -> List[int]:
        result = []
        max_height_so_far = 0

        for index in range(len(heights) - 1, -1, -1):
            current_height = heights[index]

            if current_height > max_height_so_far:
                result.append(index)
                max_height_so_far = current_height

        return result[::-1]
```

```
class NumArray:

  def __init__(self, nums: list[int]):
    self.preSum = nums
    for i in range(1, len(nums)):
      self.preSum[i] = self.preSum[i-1] + nums[i]
    
  
  
  def sumRange(self, left: int, right: int) -> int:
    if left != 0:
      return self.preSum[right] - self.preSum[left-1]
    return self.preSum[right]
```
